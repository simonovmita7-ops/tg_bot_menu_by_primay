async def handle_web_app_data(update: Update, context: ContextTypes.DEFAULT_TYPE): raw_data = update.effective_message.web_app_data.data try: data = json.loads(raw_data) data_type = data.get("type") action = data.get("action") asyncio.create_task(auto_delete(context, update.effective_message, delay=0)) if data_type == "click": if action == "connect": await connect_cmd(update, context) elif action == "reconnect": await reconnect_cmd(update, context) elif action == "disconnect": await exit_cmd(update, context) elif action == "reset": base_disconnect() base_connect() await update.effective_message.reply_text("✅ Ресет выполнен") elif action in ["lite1", "lite2", "lite3"]: open_menu_action() if action == "lite1": click_lite_server(LITE1_X, LITE1_Y) elif action == "lite2": click_lite_server(LITE2_X, LITE2_Y) elif action == "lite3": click_lite_server(LITE3_X, LITE3_Y) context.bot_data['current_server'] = f"🟢 Сервер: {action.upper()}" elif action == "graphics_max": G_G() msg = await update.effective_message.reply_text("✅ Макс. графика применена") asyncio.create_task(auto_delete(context, msg)) elif action == "graphics_min": AFK_graphics() msg = await update.effective_message.reply_text("✅ Мин. графика применена") asyncio.create_task(auto_delete(context, msg)) elif action == "esc": await esc_cmd(update, context) elif action == "screen": await screen_cmd(update, context) elif data_type == "input": text_value = data.get("text", "") if action == "chat": keyboard.press_and_release('t') time.sleep(0.3) keyboard.write(text_value) time.sleep(0.3) keyboard.press_and_release('enter') elif action == "cmd": # HTML отправляет "cmd" keyboard.press_and_release('t') time.sleep(0.3) keyboard.write('/' + text_value) time.sleep(0.3) keyboard.press_and_release('enter') except Exception as e: logging.error(f"Ошибка WebApp: {e}")
